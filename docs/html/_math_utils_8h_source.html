199 inline float BounceFunction(const float t)
200 {
201  if (t < 4.0f/11.0f)
202  {
203  return (121.0f * t * t)/16.0f;
204  }
205  else if (t < 8.0f/11.0f)
206  {
207  return (363.0f/40.0f * t * t) - (99.0f/10.0f * t) + 17.0f/5.0f;

229 // Tweens the given value (val) based on the tweening function (tweening Func defaulting to linear) and the tweening mode (tweeningMode defaulting to TweeningMode::EASE_OUT)
230 inline float TweenValue(const float val, const std::function<float(const float)> tweeningFunc = LinearFunction , const TweeningMode tweeningMode = TweeningMode::EASE_OUT)
231 {
232  switch (tweeningMode)
233  {
234  case TweeningMode::EASE_IN: return tweeningFunc(val); break;
235  case TweeningMode::EASE_OUT: return 1.0f - tweeningFunc(1.0f - val); break;
236  case TweeningMode::EASE_IN_OUT: return (val < 0.5f) ? TweenValue(val * 2.0f, tweeningFunc, TweeningMode::EASE_IN)/2.0f : 0.5f + (TweenValue((val - 0.5f) * 2.0f, tweeningFunc, TweeningMode::EASE_OUT)/2.0f);
237  }
238 
239  return 0.0f;
240 }
